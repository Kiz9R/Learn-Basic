query GetStudentTestScores($schoolName: String!, $subject: String!, $class: Int!, $section: String!, $chapterName: String!, $testName: String!) {
  chapter(where: {Students: {school_name: {_eq: $schoolName}, class: {_eq: $class}, section: {_eq: $section}}, subject: {_eq: $subject}, chapter_name: {_eq: $chapterName}, test_name: {_eq: $testName}}, order_by: {Students: {roll_number: asc}}) {
    Students {
      roll_number
      student_name
      class
      section
    }
    score
    time_taken_in_sec
    subject
    chapter_name
    test_name
  }
}
